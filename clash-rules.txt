# ============================
# Clash Meta 全规则类型
# 每条规则都是独立可用示例
# ============================

# -------- 域名类规则 --------

# 1. DOMAIN（精确匹配域名）
- DOMAIN,example.com,PROXY

# 2. DOMAIN-SUFFIX（域名后缀匹配）
- DOMAIN-SUFFIX,google.com,PROXY

# 3. DOMAIN-KEYWORD（域名关键字匹配）
- DOMAIN-KEYWORD,google,PROXY

# 4. GEOSITE（使用 geosite 数据库）
- GEOSITE,google,PROXY

# 5. RULE-SET（引用外部规则集）
- RULE-SET,google_domain,PROXY


# -------- IP 类规则 --------

# 6. IP-CIDR（IPv4 段匹配）
- IP-CIDR,1.1.1.0/24,PROXY,no-resolve

# 7. IP-CIDR6（IPv6 段匹配）
- IP-CIDR6,2404:6800::/32,PROXY,no-resolve

# 8. GEOIP（使用 geoip 数据库）
- GEOIP,US,PROXY

# 9. SRC-IP-CIDR（来源 IP 匹配）
- SRC-IP-CIDR,192.168.0.100/32,PROXY


# -------- 端口类规则 --------

# 10. DST-PORT（目标端口匹配）
- DST-PORT,443,PROXY

# 11. SRC-PORT（来源端口匹配）
- SRC-PORT,12345,PROXY


# -------- 协议类规则 --------

# 12. PROTOCOL（协议匹配）
- PROTOCOL,QUIC,PROXY

# 13. NETWORK（网络类型匹配）
- NETWORK,udp,PROXY


# -------- 进程类规则（仅支持的平台有效） --------

# 14. PROCESS-NAME（进程名匹配）
- PROCESS-NAME,chrome.exe,PROXY

# 15. PROCESS-PATH（进程路径匹配）
- PROCESS-PATH,C:\Program Files\Chrome\chrome.exe,PROXY

# 16. PROCESS（Clash Meta 新增）
- PROCESS,chrome,PROXY


# -------- 设备类规则 --------

# 17. SOURCE-IP（来源设备 IP）
- SOURCE-IP,192.168.0.101,PROXY

# 18. SOURCE-PORT（来源端口）
- SOURCE-PORT,50000,PROXY

# 19. INBOUND（入口匹配）
- INBOUND,tun,PROXY


# -------- 最终规则 --------

# 20. MATCH（兜底规则）
- MATCH,DIRECT
